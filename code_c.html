1. 進階溫度監測系統 (temp_sampler.c)

功能說明：模擬 10 次溫度採樣，判斷過熱或過低狀態，並將數據與統計摘要存入文本檔案。

#include <stdio.h>
#include <stdlib.h>
#include <time.h>

int main() {
    int n = 10;                // 採樣次數
    float temps[10];           // 儲存溫度的陣列
    float sum = 0, avg;
    float max_t, min_t;
    int i;

    // 初始化隨機數與檔案指標
    srand(time(NULL));
    FILE *fptr;
    fptr = fopen("temp_log.txt", "w"); // 以寫入模式開啟檔案

    if (fptr == NULL) {
        printf("錯誤：無法建立紀錄檔！\n");
        return 1;
    }

    printf("=== 進階監測系統：開始 10 次數據採樣 ===\n\n");
    fprintf(fptr, "時間點,溫度(C),狀態\n"); // 寫入檔案標題

    for (i = 0; i < n; i++) {
        // 模擬溫度 15.0 ~ 45.0
        temps[i] = (rand() % 301 / 10.0) + 15.0;
        
        // 初始極值
        if (i == 0) {
            max_t = temps[i];
            min_t = temps[i];
        } else {
            if (temps[i] > max_t) max_t = temps[i];
            if (temps[i] < min_t) min_t = temps[i];
        }

        sum += temps[i];

        // 判斷狀態並顯示於螢幕與寫入檔案
        const char *status;
        if (temps[i] >= 35.0) status = "過熱";
        else if (temps[i] <= 20.0) status = "過低";
        else status = "正常";

        printf("採樣 [%d]: %.1f °C (%s)\n", i + 1, temps[i], status);
        fprintf(fptr, "%d,%.1f,%s\n", i + 1, temps[i], status);
    }

    avg = sum / n;

    // 輸出統計結果
    printf("\n--- 統計報告 ---\n");
    printf("平均溫度: %.2f °C\n", avg);
    printf("最高溫度: %.1f °C\n", max_t);
    printf("最低溫度: %.1f °C\n", min_t);
    printf("----------------\n");
    printf("所有數據已儲存 至 temp_log.txt\n");

    // 將統計摘要也存入檔案
    fprintf(fptr, "\n[統計摘要]\n平均:%.2f\n最大:%.1f\n最小:%.1f\n", avg, max_t, min_t);

    fclose(fptr); // 關閉檔案
    return 0;
}

2. 歐姆定律與電阻計算機 (resistor_calc.c)

功能說明：計算兩個電阻的串聯或並聯總阻值，並結合歐姆定律計算總電流。

#include <stdio.h>

int main() {
    int choice;
    float r1, r2, total_r, voltage, current;

    printf("=== 歐姆定律與電阻計算 ===\n");
    printf("1. 計算串聯總電阻 (Series)\n");
    printf("2. 計算並聯總電阻 (Parallel)\n");
    printf("請選擇計算模式 (1 或 2): ");
    if (scanf("%d", &choice) != 1) return 1;

    printf("請輸入第一個電阻值 (R1, 歐姆): ");
    scanf("%f", &r1);
    printf("請輸入第二個電阻值 (R2, 歐姆): ");
    scanf("%f", &r2);

    // 邏輯判斷與計算
    if (choice == 1) {
        total_r = r1 + r2;
        printf("\n[計算結果] 串聯總電阻為: %.2f Ohm\n", total_r);
    } 
    else if (choice == 2) {
        if (r1 + r2 == 0) {
            printf("\n錯誤：電阻值不能同時為 0！\n");
            return 1; 
        }
        total_r = (r1 * r2) / (r1 + r2);
        printf("\n[計算結果] 並聯總電阻為: %.2f Ohm\n", total_r);
    } 
    else {
        printf("\n選項無效，請重新執行程式。\n");
        return 1;
    }

    // 進階功能：結合歐姆定律計算電流
    printf("\n--- 進階電流計算 --- \n");
    printf("請輸入供應電壓 (V, 伏特): ");
    scanf("%f", &voltage);

    if (total_r > 0) {
        current = voltage / total_r;
        printf("在 %.2fV 下，總電流為: %.4f A (安培)\n", voltage, current);
    } else {
        printf("總電阻為 0，無法計算電流。\n");
    }

    printf("========================================\n");
    return 0;
}
