1. 進階溫度監測系統 (temp_sampler.c)

功能：模擬溫度採樣、判斷狀態並將統計數據輸出至檔案。

#include <stdio.h>
#include <stdlib.h>
#include <time.h>

int main() {
    int n = 10;
    float temps[10], sum = 0, avg, max_t, min_t;
    int i;

    srand(time(NULL));
    FILE *fptr = fopen("temp_log.txt", "w");
    if (fptr == NULL) return 1;

    fprintf(fptr, "時間點,溫度(C),狀態\n");
    for (i = 0; i < n; i++) {
        temps[i] = (rand() % 301 / 10.0) + 15.0;
        if (i == 0) { max_t = min_t = temps[i]; }
        else {
            if (temps[i] > max_t) max_t = temps[i];
            if (temps[i] < min_t) min_t = temps[i];
        }
        sum += temps[i];

        const char *status = (temps[i] >= 35.0) ? "過熱" : (temps[i] <= 20.0 ? "過低" : "正常");
        printf("採樣 [%d]: %.1f °C (%s)\n", i + 1, temps[i], status);
        fprintf(fptr, "%d,%.1f,%s\n", i + 1, temps[i], status);
    }
    avg = sum / n;
    printf("\n平均: %.2f | 最高: %.1f | 最低: %.1f\n", avg, max_t, min_t);
    fclose(fptr);
    return 0;
}

2. 歐姆定律與電阻計算機 (resistor_calc.c)

功能：支援串聯與並聯總電阻計算，並整合歐姆定律推算電流。

#include <stdio.h>

int main() {
    int choice;
    float r1, r2, total_r, voltage, current;

    printf("1. 串聯 (Series) | 2. 並聯 (Parallel): ");
    if (scanf("%d", &choice) != 1) return 1;

    printf("輸入 R1, R2: ");
    scanf("%f %f", &r1, &r2);

    if (choice == 1) {
        total_r = r1 + r2;
    } else {
        if (r1 + r2 == 0) return 1;
        total_r = (r1 * r2) / (r1 + r2);
    }
    printf("總電阻: %.2f Ohm\n", total_r);

    printf("輸入電壓 (V): ");
    scanf("%f", &voltage);
    if (total_r > 0) {
        printf("總電流: %.4f A\n", voltage / total_r);
    }
    return 0;
}
